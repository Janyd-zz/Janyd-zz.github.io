#从docker hub拉取镜像，repository为ubuntu,TAG命名为18.04
docker pull ubuntu:18.04

#查看容器镜像
docker images

#查看正在运行的容器
docker container ps

#查看所有的容器,包含正在运行的和不在运行的
docker container ps -a

#查看容器的大小
docker container ps -a --size

#创建容器
docker run -it --name my-ubuntu-18.04 --hostname hello ubuntu:18.04

#创建一个容器，-v将主机/home/janyd/readme挂载到容器的/root/readme
docker run -it -v /home/janyd/readme:/root/readme --name ubuntu-data --hostname data ubuntu:18.04

#退出运行的容器
exit

#重新运行容器
docker start ubuntu-data

#进入该容器
docker exec -it ubuntu-data bash

#将容器打包导出来
exit
docker export ubuntu-data > ubuntu-data.tar

#将容器导入,TAG命名为为data
docker import ubuntu-data.tar ubuntu:data

#将ubuntu:data的镜像commit到ubuntu:data-v1
docker commit ubuntu-data ubuntu-data-v1.0

#将镜像save出去,镜像可以记录历史commit
docker save ubuntu:data > ubuntu-data-image.tar

#导入镜像
docker load -i ubuntu-data-image.tar

#删除镜像
docker image rm ubuntu:data

#删除容器
docker container rm my-ubuntu-18.04
